data = input().split()
a = int(data[0])
b = int(data[1])

orig_a = a
orig_b = b

x0, y0 = 1, 0  
x1, y1 = 0, 1   

while b != 0:
    q = a // b
    
    temp_a = a
    temp_b = b
    
    a = b
    b = temp_a % temp_b
    
    temp_x0 = x0
    temp_y0 = y0
    
    x0 = x1
    y0 = y1
    
    x1 = temp_x0 - q * x1
    y1 = temp_y0 - q * y1
d = a
x = x0
y = y0
print(x, y, d)
